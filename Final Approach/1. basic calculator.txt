start the program
    input operation character
    input var1 var2 result integer
    if operation == '+'
        assign result = var1 + var2
    else if operation == '-'
        assign result = var1 - var2
    else if operation == '*'
        assign result = var1 * var2
    else if operation == '/'
        assign result = var1 / var2
    end if
    print variable result 
end
