# calculator program +-*/
start the program
    input operation character
    input var1 and var2 integer
    if operation is +
        print var1 + var2
    else if operation is -
        print var1 - var2
    else if operation is *
        print var1 * var2
    else if operation is /
        print var1 / var2
    end if
end

# square root of given number
